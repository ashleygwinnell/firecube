ofstream file;
void Logger::Init(string filename)
{
    file.open(filename.c_str(),ios::out|ios::trunc);    
}
void Logger::Write(string str)
{
    if (file.is_open())
        file << str;
}
void Logger::Close()
{
    file.close();
}
